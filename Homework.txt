using System; 
using System.Collections.Generic;
using System.Drawing;
using System.IO;
using System.Text.Json;
using System.Windows.Forms;

namespace FormOptions
{
    public partial class MainForm : Form
    {
        bool isDarkMode = false; //Змінна для збереження теми

        string configPath = "appsettings.json"; //Шлях до файлу з налаштуваннями

        private int currentQuestionIndex = 0; //Поточний індекс до кожного запитання

        private int score = 0; //Показує кількість правильних відповідей

        private List<Question> questions = new List<Question> //Список запитань у тесті
        {
            new Question {
                Text = "Яка подія вважається початком Другої світової війни?",
                Answers = new[] { "Напад на Польщу", "Аншлюс Австрії", "Перл-Гарбор", "Пакт М-Р" },
                CorrectAnswerIndex = 0
            },
            new Question {
                Text = "Як називався німецький план нападу на СРСР?",
                Answers = new[] { "Вайс", "Барбаросса", "Оверлорд", "Гельб" },
                CorrectAnswerIndex = 1
            },
            new Question {
                Text = "В якому році відбулася висадка в Нормандії?",
                Answers = new[] { "1941", "1943", "1944", "1945" },
                CorrectAnswerIndex = 2
            },
            new Question {
                Text = "Що намалювали вандали на електричці?",
                Answers = new[] { "Свастику", "Смайлик", "ЦОЙ ЖИВ", "Інше" },
                CorrectAnswerIndex = 0
            }
        };

        public MainForm() //Конструктор форми
        {
            InitializeComponent();

            //Підписка на зміну стану радіокнопок
            rdBtnAnswer1.CheckedChanged += AnyRadioButton_CheckedChanged;
            rdBtnAnswer2.CheckedChanged += AnyRadioButton_CheckedChanged;
            rdBtnAnswer3.CheckedChanged += AnyRadioButton_CheckedChanged;
            rdBtnAnswer4.CheckedChanged += AnyRadioButton_CheckedChanged;
        }

        private void MainForm_Load(object sender, EventArgs e) //Подія для завантаження форми
        {
            string str = Newtonsoft.Json.JsonConvert.SerializeObject( //Серіалізація списку запитань у JSON файл
                questions,
                Newtonsoft.Json.Formatting.Indented
            );

            File.WriteAllText("questions.json", str);
            LoadSettings(); //Завантажує налаштування
            ApplyTheme(); //Застосує теми
            DisplayQuestion(); //Відображає перше запитання
        }

        private void DisplayQuestion() //Це метод для відображення запитання
        {
            if (currentQuestionIndex < questions.Count) //Використовується, якщо ще є питання
            {
                var q = questions[currentQuestionIndex];

                //Виводить текст запитання і відповіді
                lbQuestion.Text = q.Text;
                rdBtnAnswer1.Text = q.Answers[0];
                rdBtnAnswer2.Text = q.Answers[1];
                rdBtnAnswer3.Text = q.Answers[2];
                rdBtnAnswer4.Text = q.Answers[3];

                //Скидання вибору
                rdBtnAnswer1.Checked = false;
                rdBtnAnswer2.Checked = false;
                rdBtnAnswer3.Checked = false;
                rdBtnAnswer4.Checked = false;

                btnCont.Enabled = false; //Кнопка "Далі" неактивна, поки не вибереш відповідь
            }
            else
            {
                //Якщо питання закінчились, показує результат тесту
                MessageBox.Show(
                    $"Тест закінчено! Правильних відповідей: {score} з {questions.Count}",
                    "Результат"
                );

                //Скидання тесту
                currentQuestionIndex = 0;
                score = 0;

                DisplayQuestion();
            }
        }

        private void AnyRadioButton_CheckedChanged(object sender, EventArgs e) //Обробник вибору відповіді
        {
            //Якщо якась кнопка вибрана, натискаємо кнопку "Далі"
            if (sender is RadioButton rb && rb.Checked)
            {
                btnCont.Enabled = true;
            }
        }

        private void btnCont_Click(object sender, EventArgs e) //кнопка "Далі"
        {
            RadioButton[] answers = { rdBtnAnswer1, rdBtnAnswer2, rdBtnAnswer3, rdBtnAnswer4 };
            int selectedIndex = -1;

            for (int i = 0; i < answers.Length; i++) //Знаходить обрану відповідь
            {
                if (answers[i].Checked)
                {
                    selectedIndex = i;
                    break;
                }
            }

            if (selectedIndex == questions[currentQuestionIndex].CorrectAnswerIndex) //Перевірка правильних відповідей
            {
                score++;
            }

            //Переходить до наступного питання
            currentQuestionIndex++;
            DisplayQuestion();
        }

        private void btnChangeStyles_Click(object sender, EventArgs e) //Кнопка зміни теми
        {
            isDarkMode = !isDarkMode; //Перемикає тему

            ApplyTheme();
            SaveSettings();
        }

        private void ApplyTheme() //Метод застосування теми
        {
            bool dark = isDarkMode;

            this.BackColor = dark ? Color.FromArgb(26, 26, 26) : SystemColors.Control; //Змінює фону форми

            foreach (Control ctrl in this.Controls) //Проходить по всіх елементах
            {
                if (ctrl is Label || ctrl is RadioButton) //Для тексту
                    ctrl.ForeColor = dark ? Color.White : Color.Black;

                if (ctrl is Button btn) //Для кнопок
                {
                    btn.BackColor = dark ? Color.DimGray : Color.White;
                    btn.ForeColor = dark ? Color.White : Color.Black;
                }
            }

            btnChangeStyles.Text = dark ? "Світла" : "Темна"; //Текст для кнопки перемикання теми
        }

        private void LoadSettings() //Завантажує налаштування з файлу
        {
            try
            {
                if (File.Exists(configPath))
                {
                    string jsonString = File.ReadAllText(configPath);

                    using (JsonDocument doc = JsonDocument.Parse(jsonString))
                    {
                        isDarkMode = (doc.RootElement //Перевіряє теми
                            .GetProperty("theme")
                            .GetString() == "dark");
                    }
                }
            }
            catch { } //Ігнорує всі помилки
        }

        //Зберігає налаштування у файл
        private void SaveSettings()
        {
            try
            {
                var data = new
                {
                    theme = isDarkMode ? "dark" : "light"
                };

                string jsonString = JsonSerializer.Serialize(
                    data,
                    new JsonSerializerOptions { WriteIndented = true }
                );

                File.WriteAllText(configPath, jsonString);
            }
            catch { } //Ігнорує всі помилки
        }
    }
}
